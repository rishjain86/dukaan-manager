rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /shops/{shopId}/items/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null && get(/databases/(default)/documents/shopUsers/$(request.auth.uid)).data.shopId == shopId;
    }
    match /companion/{userId}/{allPaths=**} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
